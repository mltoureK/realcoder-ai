export function extendSourcesWithPercentages({
    processedData,
    totalVisitors,
    mode
}: ExtendSourcesOptions): ProcessedSourceData[] {
    if (mode === 'growth') {
        return processedData;
    }

    return processedData.map(item => ({
        ...item,
        percentage: totalVisitors > 0 ? (item.visits / totalVisitors) : 0
    }));
};
